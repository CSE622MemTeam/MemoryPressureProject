049    /**Searches for current oom-value and system state*/
050    /**We can better predict the chances by monitoring the file pages in the zone than monitoring free*/
051    public static void updateMemoryStatus() {
052        Iterator<Integer> iterator = oom_score.iterator();
053        long expectedFreePages=0;
054        int key = 0, value=0;
055        
056        while(iterator.hasNext())
057        {
058            if(!((value=iterator.next())<oom_adj))
059            {
060                /**Searching for a match*/
061                key = value;
062                break;
063            }
064        }
065        
066        if (oom_map.size() == 0)
067            buildTable();
068        
069        expectedFreePages = oom_map.get(key);
070                
071        if(nrFreePages< expectedFreePages) {
072            memory_state = MEMORY_CRITICAL;
073        }
074        else if(nrFreePages< (expectedFreePages-warningAt)) {
075            memory_state = MEMORY_LOW;
076        }
077        else {
078            memory_state = MEMORY_OK;
079        }
080        
081        dumpString();
082    }

084    /**update and print values to csv file*/
085    public static void dumpString() {
086        FileOperations.createFile();
087        pid = android.os.Process.myPid();        
088        vmUsage      = scanProcForField("/proc/"+pid+"/status", "VmRSS:"); 
089        memFree      = scanProcForField("/proc/meminfo", "MemFree:");
090        totMem       = scanProcForField("/proc/meminfo", "MemTotal:");
091        nrFreePages  = scanProcForField("/proc/zoneinfo", "nr_file_pages");
092        oom_adj      = scanProcForFieldInt("/proc/"+pid+"/oom_adj");
093        Runtime info = Runtime.getRuntime(); 
094        max_memory   = info.maxMemory();
095        cur_memory   = (info.totalMemory() - info.freeMemory());
096        
097        String dumpData = pid+","+vmUsage + "," + memFree +","+totMem+","+nrFreePages+","+oom_adj+","+cur_memory+","+memory_state+"\n";
098
099        try {
100            FileOperations.write(dumpData);
101        } catch (IOException e) {
102            /**Error writing*/
103            Log.e("cse_622", "Cannot write values to file");
104            e.printStackTrace();
105        }
106    }

108    /**scan and build the low memory killer parameter table*/
109    public static void buildTable() {
110        try {
111            scanMemoryKiller("/sys/module/lowmemorykiller/parameters/minfree","/sys/module/lowmemorykiller/parameters/adj");
112        } catch (IOException e) {
113            Log.e("cse_622", "Cannot read oom specification file");
114            e.printStackTrace();
115            return;
116        }
117        
118        if(free_pages.isEmpty()||oom_score.isEmpty()||(free_pages.size()!=oom_score.size())) {
119            Log.e("cse_622", "Something has wen't wrong. OOM table size's mismatch");
120            return;
121        }
122        else {
123            Iterator<Integer> iterator  = free_pages.iterator();
124            Iterator<Integer> iterator1 = oom_score.iterator();
125            
126            while(iterator.hasNext() && iterator1.hasNext()) {                                
127                int key   = iterator1.next();
128                int value = iterator.next();
129                
130                oom_map.put(key, value);
131            }
132        }  
133    }

175    /** 
176     *  The available memory on the system. This number should not be considered absolute: due 
177     *  to the nature of the kernel, a significant portion of this memory is actually in use and 
178     *  needed for the overall system to run well.
179     */
180    public static long getAvailableMem(Context context) {
181        ActivityManager activityManager =  (ActivityManager) context.getSystemService(Context.ACTIVITY_SERVICE);
182        MemoryInfo memoryInfo = new MemoryInfo();
183        activityManager.getMemoryInfo(memoryInfo);
184        return memoryInfo.availMem;
185    }

187    /** 
188     *  The threshold of availMem at which we consider memory to be low and start killing

192        ActivityManager activityManager =  (ActivityManager) context.getSystemService(Context.ACTIVITY_SERVICE);
193        MemoryInfo memoryInfo = new MemoryInfo();
194        activityManager.getMemoryInfo(memoryInfo);
195        return memoryInfo.threshold;

198    /** Pretty version of bytes used for printing. */
199    public static String bytesToString(long bytes) {
200        if (bytes < 1024)
201            return bytes+"B";
202        if ((bytes>>10) < 1024)
203            return (bytes>>10)+"KB";
204        if ((bytes>>20) < 1024)
205            return (bytes>>20)+"MB";
206        return (bytes>>30)+"GB";
207    }

209    private static long scanProcForField(String path, String field) {
210        File file = new File(path);
211        Scanner scanner = null;
212
213        try {
214            scanner = new Scanner(file);
215            while (scanner.hasNextLine()) {
216                String token = scanner.next();
217                if (token.equals(field))
218                    return scanner.nextLong() << 10; 
219                scanner.nextLine();
220            }
221        } catch (Exception e) {
222            Log.e("MemoryUtil", "Error scanning " + file + " for " + field, e);
223        } finally {
224            if (scanner != null)
225                scanner.close();
226        }
227        return -1;
228    }
229    
230    private static int scanProcForFieldInt(String path) {
231        File file = new File(path);
232        Scanner scanner = null;
233
234        try {
235            scanner = new Scanner(file);
236            while (scanner.hasNextLine()) {
237                return scanner.nextInt(); 
238            }
239        } catch (Exception e) {
240            Log.e("MemUtil", "Error scanning " + file, e);
241        } finally {
242            if (scanner != null)
243                scanner.close();
244        }
245        return -1;
246    }

248    private static void scanMemoryKiller(String path_free_pages, String path_score) throws IOException {
249        FileReader input  = new FileReader(path_free_pages);
250        FileReader input1 = new FileReader(path_score);
251    
252        BufferedReader buf = new BufferedReader(input);
253        String           temp = null;
254    
255        while((temp = buf.readLine())!=null) {
256            String[] stringFreePages = temp.split(",");
257
258            for(int i=0; i<stringFreePages.length; i++) {
259                free_pages.add(Integer.parseInt(stringFreePages[i]));
260            }
261        }
262    
263        temp = null;
264        buf.close();
265    
266        BufferedReader buf1 = new BufferedReader(input1);
267    
268        while((temp = buf1.readLine())!=null) {
269            String stringScore[] = temp.split(",");
270        
271            for(int j=0; j<stringScore.length; j++) {
272                oom_score.add(Integer.parseInt(stringScore[j]));
273            }
274        }
275    
276        buf1.close();
277    }
