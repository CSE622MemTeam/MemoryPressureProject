001package edu.buffalo.memlib.manager;
002
003import java.io.BufferedWriter;
004import java.io.File;
005import java.io.FileWriter;
006import java.io.IOException;
007import java.text.SimpleDateFormat;
008import java.util.Date;
009import java.util.Locale;
010
011import android.os.Environment;
012import android.util.Log;
013
014public final class FileOperations {
015    
016    public static String sd_path;
017    public static int    pid= 0;
018    public static String file_path=null;
019    
020    public FileOperations()
021    {
022        
023    }
024        
025    /**creates a time-stamped csv file. Call before starting the thread*/
026    public static Boolean createFile() {
027        if (file_path != null)
028            return true;
029        
030        SimpleDateFormat fileDate = new SimpleDateFormat("ddMMyyyyhhmmss", Locale.US);
031        
032        String timeStamp = fileDate.format(new Date());
033        
034        pid       = android.os.Process.myPid();
035        sd_path   = Environment.getExternalStorageDirectory().toString();        
036        file_path = sd_path+"/memory_usage_"+timeStamp+".csv";
037        File file = new File(file_path);
038        
039        if(!file.exists())
040        {
041            try {
042                file.createNewFile();
043                Log.d("cse622_debug", "Created a file /mnt/sdcard/");
044            } catch (IOException e) {
045                Log.e("cse622_error", "Cannot create file memory_usage");
046                e.printStackTrace();
047            }
048        }
049        
050        return writeHeaders();
051    }
052    
053    /**Writes the column headers for the csv file.Call before starting the thread*/
054    public static Boolean writeHeaders()
055    {   
056        String Headers = "PID, VMRss, MemFree, MemTotal, FreePages, oomValue, curMem(in bytes), state\n";
057        
058        try {
059            FileOperations.write(Headers);
060        } catch (IOException e) {
061            Log.e("cse622_error", "Cannot write to file memory_usage");
062            e.printStackTrace();
063        }
064        
065        return true;
066        
067    }
068    
069    /**Write anything to the csv file*/
070    public static Boolean write(String content) throws IOException
071    {
072        File file        = new File(file_path);    
073            
074        try {
075             FileWriter filewriter = new FileWriter(file.getAbsoluteFile(),true);
076             BufferedWriter bufferedwriter = new BufferedWriter(filewriter);
077             
078             bufferedwriter.write(content);
079             
080             bufferedwriter.close();
081             return true;
082             
083        } catch (IOException e) {
084            Log.e("cse622_error", "Cannot Open file memory_usage");
085            e.printStackTrace();
086            return false;
087        }
088        
089        
090    }
091    
092}
