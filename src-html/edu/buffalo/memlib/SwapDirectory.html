001package edu.buffalo.memlib;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.ObjectInputStream;
008import java.io.ObjectOutputStream;
009
010/**
011 * All swap-related operations and configurations are performed through this
012 * class. This class should only be used internally.
013 */
014final class Swap {
015    /** The default base name of the swap directory. */
016    // TODO: Make this configurable.
017    static String swapDirName = ".swap";
018
019    /** Swap an object out and return its swap token. */
020    static synchronized <T> Token<T> swapOut(T t) throws IOException {
021        Token<T> token = new Token<T>();
022        SwapDirectory.open(token, true).swapOut(t);
023        return token;
024    }
025
026    /** Swap the object with the given swap token back in. */
027    static synchronized <T> T swapIn(Token<T> token) throws IOException, ClassNotFoundException {
028        T t = SwapDirectory.open(token, false).swapIn();
029        SwapDirectory.free(token);
030        return t;
031    }
032
033    /** Free a swapped out object, given its token. */
034    static synchronized void free(Token<?> token) throws IOException {
035        SwapDirectory.free(token);
036    }
037
038    /** Configure the location for the swap directory. */
039    static synchronized void setRoot(String path) {
040        setRoot(new File(path));
041    }
042
043    /** Configure the location for the swap directory. */
044    static synchronized void setRoot(File dir) {
045        dir = new File(dir, swapDirName);
046        SwapDirectory.directory(dir);
047    }
048
049    /** A token which can be used to recover a swapped object. */
050    static class Token<T> {
051        private static volatile long nextTokenId = 0;
052        final long value = nextTokenId();
053
054        private static synchronized long nextTokenId() {
055            return nextTokenId++;
056        }
057
058        public String toString() {
059            return Long.toHexString(value);
060        }
061    }
062
063    private Swap() { /* Don't make me. */ }
064}
065
066/**
067 * A file system location where swapped objects go. For now, the members of
068 * this class are static for simplicity's sake. In the future, it should be
069 * possible to manage multiple swap directories with multiple SwapDirectory
070 * instances.
071 */
072final class SwapDirectory {
073    /** The current swap location. Lazily initialized. */
074    private static File directory;  // XXX: Use carefully.
075
076    /** Open a new swap file for token. */
077    static <T> SwapFile<T> open(Swap.Token<T> token, boolean create) throws IOException {
078        File file = fileFor(token);
079        return new SwapFile<T>(file, create);
080    }
081
082    /** Delete the swap file for the given token. */
083    static void free(Swap.Token<?> token) {
084        fileFor(token).delete();
085    }
086
087    /** Clear all the files in the swap directory. */
088    static void clear() {
089        for (File file : directory().listFiles()) file.delete();
090    }
091
092    /** Get the file associated with the token. */
093    static private File fileFor(Swap.Token<?> token) {
094        return new File(directory(), token.toString());
095    }
096
097    /** Get the swap directory. If none is set, a default will be chosen. */
098    static synchronized File directory() {
099        return directory(directory);
100    }
101
102    /** Set the swap directory. If dir is null, a default will be chosen. */
103    static synchronized File directory(File dir) {
104        return directory = (dir != null) ? dir : defaultDirectory();
105    }
106
107    /** Create and return a default swap directory location. */
108    private static synchronized File defaultDirectory() {
109    // TODO: Improve how we decide where the default swap directory should go.
110    // This could involve finding the running app's private application storage
111    // directory, finding a suitable location in internal or external storage,
112    // or some other method.  For now, just create and return a new temp
113    // directory.
114        try {
115            File tmp = File.createTempFile(Swap.swapDirName, null);
116            tmp.delete();
117            tmp.mkdirs();  // FIXME: Possible race condition...
118            return tmp;
119        } catch (Exception e) {
120            throw new RuntimeException(e);
121        }
122    }
123}
124
125/** A handle on an individual swap file. */
126final class SwapFile<T> {
127    private final File file;
128
129    /** Create and open file with mode. */
130    public SwapFile(File file, boolean create) throws IOException {
131        if (create && !file.createNewFile())
132            throw new RuntimeException("Could not create file.");
133        if (!create && !file.exists())
134            throw new RuntimeException("File does not exist");
135        this.file = file;
136    }
137
138    /** Get the object stored in the swap file. */
139    public synchronized T swapIn() throws IOException, ClassNotFoundException {
140        ObjectInputStream ois = new ObjectInputStream(new FileInputStream(file));
141        @SuppressWarnings("unchecked")
142        T t = (T) ois.readObject();
143        ois.close();
144        return t;
145    }
146
147    /** Put an object into the swap file. */
148    public synchronized void swapOut(T t) throws IOException {
149        ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(file));
150        oos.writeObject(t);
151        oos.close();
152    }
153}
