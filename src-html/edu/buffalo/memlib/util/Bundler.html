042/** Put utility methods here until we can find a better home. */
043class Util {
044        /** Used to fix primitive classes. */
045        static final Map<Class,Class> PRIM_MAP = new HashMap<Class,Class>() {{
046                put(boolean.class, Boolean.class);
047                put(char.class, Character.class);
048                put(byte.class, Byte.class);
049                put(short.class, Short.class);
050                put(int.class, Integer.class);
051                put(long.class, Long.class);
052                put(float.class, Float.class);
053                put(double.class, Double.class);
054        }};
055
056        static Class<?> normalizeType(Class<?> clazz) {
057                if (clazz.isPrimitive())
058                        clazz = PRIM_MAP.get(clazz);
059                return clazz;
060        }
061
062        /** Check if serializable, or array of serializables. */
063        static boolean isSerializable(Object o) {
064                Class<?> component = o.getClass();
065
066                // Find component type, if array.
067                while (component.isArray())
068                        component = component.getComponentType();
069
070                return Serializable.class.isAssignableFrom(component);
071        }
072
073        static boolean shouldIgnoreField(Field f) {
074                int mod = 0;//f.getModifiers();
075                return Modifier.isTransient(mod) || Modifier.isStatic(mod);
076        }
077}

079/**
080 * A wrapper around a Bundle that provides some nicer methods. This is done as
081 * a decorator because Bundle is final. This should be used only by MLBundler
082 * and MLUnbundler below.
083 */
084final class MLBundle {
085        final Bundle bundle;
086
087        MLBundle() {
088                this(new Bundle());
089        }
090
091        MLBundle(Bundle bundle) {
092                this.bundle = bundle;
093        }
094
095        /**
096         * Add an object to the bundle. This calls the appropriate bundle method
097         * based on object type. Returns the encapsulated Bundle.
098         */
099        Bundle put(String key, Object o) {
100                if (o == null)
101                        return bundle;
102                //else if (o instanceof Parcelable)
103                        //bundle.putParcelable(key, (Parcelable) o);
104                else if (Util.isSerializable(o))
105                        bundle.putSerializable(key, (Serializable) o);
106                else
107                        throw new RuntimeException("Unable to marshal: "+o);
108                return bundle;
109        }
110}

112/** Class to convert an object into a custom bundle. */
113class MLBundler {
114        /** Path of this bundler in the hierarchy. */
115        protected String path = "<root>";
116        /** Top-most class to marshal. */
117        protected Class<?> base;
118        /** Class we're currently marshalling. */
119        protected Class<?> clazz;
120        /** Bundle we're currently working with. */
121        private MLBundle bundle = new MLBundle();
122        /** Map of visited objects. For loop detection. */
123        protected Map<Object,String> visitMap = new HashMap<Object,String>();
124
125        /** Create a bundler that will marshal everything. */
126        public MLBundler() { this(Object.class); }
127
128        /** Create a bundler that will not marshal past base. */
129        public MLBundler(Class<?> base) { this.base = base; }
130
131        /** A sub-bundler for bundling field values. */
132        class SubMLBundler extends MLBundler {
133                public SubMLBundler(String name, Class<?> clazz) {
134                        super(MLBundler.this.base);
135                        path = MLBundler.this.path+"."+name;
136                        this.clazz = clazz;
137                        visitMap = MLBundler.this.visitMap;
138                }
139        }
140
141        /** Bundle the given object. */
142        public Bundle bundle(Object object) {
143                if (object == null)
144                        return Bundle.EMPTY;
145
146                if (clazz == null)
147                        clazz = object.getClass();
148
149                String path = visitMap.get(new VKey(object));
150
151                if (path != null)
152                        return bundle.put("<ref>", path);
153
154                return visit(object);
155        }
156
157        /** Visit an object for the first time. */
158        private Bundle visit(Object object) {
159                visitMap.put(new VKey(object), path);
160
161                if (Util.isSerializable(object))
162                        return bundle.put("<data>", object);
163
164                bundle.put("<class>", object.getClass());
165
166                return putFields(object);
167        }
168
169        /** Bundle the fields of an object by name. */
170        private Bundle putFields(Object object) {
171                for (Field field : clazz.getDeclaredFields())
172                        putField(object, field);
173                return putSuperclassFields(object);
174        }
175
176        /** Bundle a single field. */
177        private void putField(Object object, Field field) {
178                boolean access = field.isAccessible();
179
180                if (!Util.shouldIgnoreField(field)) try {
181                        field.setAccessible(true);
182
183                        String key = field.getName();
184                        Object value = field.get(object);
185
186                        if (value == null)
187                                return;
188
189                        Bundle sb = new SubMLBundler(key, null).bundle(value);
190
191                        bundle.put(key, sb);
192                } catch (Exception e) {
193                        // SecurityManager doesn't let us do this. What can we do?
194                } finally {
195                        field.setAccessible(access);
196                }
197        }
198
199        /** Bundle all the fields in the superclass. */
200        private Bundle putSuperclassFields(Object object) {
201                Class<?> sc = clazz.getSuperclass();
202                if (!classIsAllowed(sc))
203                        return bundle.bundle;
204                Bundle sb = new SubMLBundler("<super>", sc).bundle(object);
205                return bundle.put("<super>", sb);
206        }
207
208        /**
209         * Make sure we're only bundling allowed classes. The first check ensures
210         * that clazz is a type of base, and the second check ensures that clazz is a
211         * strict subtype of base.
212         */
213        private boolean classIsAllowed(Class<?> clazz) {
214                return base.isAssignableFrom(clazz) && !clazz.isAssignableFrom(base);
215        }
216}

218/** Wrapper which compares object identity. Used to detect loops. */
219final class VKey {
220        final Object object;
221
222        public VKey(Object object) { this.object = object; }
223
224        public boolean equals(Object o) {
225                return (o instanceof VKey) && object == ((VKey) o).object;
226        }
227
228        public int hashCode() {
229                return System.identityHashCode(object);
230        }
231}
