001package edu.buffalo.memlib;
002
003import android.app.Activity;
004import android.content.Intent;
005import android.os.Bundle;
006import android.view.View;
007import android.widget.Button;
008
009public class MainActivity extends Activity {
010    
011    protected void onCreate(Bundle savedInstanceState) {
012        super.onCreate(savedInstanceState);
013
014        setContentView(R.layout.launcher);
015
016        Button classic = (Button) findViewById(R.id.button1);
017        classic.setOnClickListener(new View.OnClickListener() {
018            public void onClick(View view) {
019                Intent intent = new Intent(MainActivity.this, ClassicTesterActivity.class);
020                startActivity(intent);
021            }
022        });
023        
024        Button debug = (Button)findViewById(R.id.button2);
025        debug.setOnClickListener(new View.OnClickListener() {
026            public void onClick(View v) {
027                Intent intent = new Intent(MainActivity.this, DebugActivity.class);
028                startActivity(intent);
029            }
030        });
031    }
032}
