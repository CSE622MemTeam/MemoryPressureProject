001package edu.buffalo.memlib;
002
003import java.lang.reflect.InvocationHandler;
004import java.lang.reflect.Method;
005import java.lang.reflect.Proxy;
006import java.util.ArrayList;
007import java.util.HashMap;
008import java.util.HashSet;
009import java.util.LinkedList;
010import java.util.List;
011import java.util.Map;
012import java.util.Set;
013
014public class SwapObjects<T> implements InvocationHandler {
015    private SwapReference<T> swapReference;
016
017    /** initialization without object. Don't know any useful case for this
018     * for now. Requires extra function(s) to make this useful 
019     */
020    public SwapObjects() {
021        swapReference = new SwapReference<T>();
022    }
023
024    /** initialization with object. Will initialize SwapReference */
025    public SwapObjects(T object) {
026        swapReference = new SwapReference<T>(object);
027    }
028
029    /** Create proxy object which proxies Object o which implements Class c */
030    @SuppressWarnings("unchecked")
031    public static <T> T getOne(Class<?> c, T o) {
032        return (T) Proxy.newProxyInstance(SwapObjects.class.getClassLoader(),
033                                          new Class[] {c},
034                                          new SwapObjects<T>(o));
035    }
036
037    // Creating an ArrayList proxy
038    public static <T> List<T> getArrayList(){
039        return (List<T>) getOne(List.class, new ArrayList<T>());
040    }    
041
042    // Creating LinkedList proxy
043    public static <T> List<T> getLinkedList(){
044        return (List<T>) getOne(List.class, new LinkedList<T>());
045    }
046
047    // Creating HashMap proxy
048    public static <K,V> Map<K,V> getHashMap(){
049        return (Map<K,V>) getOne(Map.class, new HashMap<K,V>());
050    }
051
052    // Creating Set proxy
053    public static <T> Set<T> getHashSet(){
054        return (Set<T>) getOne(Set.class, new HashSet<T>());
055    }
056
057    /** 
058     * Any function call will pass through here and swapReference will swap in if the 
059     * object is swapped out 
060     */    
061    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
062        Object object = swapReference.get();
063        if(object != null)
064            return method.invoke(object, args);
065        return null;
066    }
067}
