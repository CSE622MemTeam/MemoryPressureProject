001package edu.buffalo.memlib;
002
003import edu.buffalo.memlib.manager.MemoryUtil;
004
005/**
006 * Application-wide daemon thread. Periodically monitors heap utilization and
007 * swaps objects as necessary.
008 */
009final class SwapManager {
010    /** Set to true when daemon thread has started. */
011    private static boolean initialized = false;
012
013    /** Start this thread to start monitoring. */
014    private static Thread daemon = new Thread("swap-manager") {
015        public void run() { monitor(); }
016    };
017
018    private static Policy policy = new Policy();
019
020    /** Set the global policy used by the swap manager. */
021    public static synchronized void setPolicy(Policy policy) {
022        SwapManager.policy = policy;
023    }
024
025    /** Initialize the swap manager thread. No effect if already started. */
026    public static void initialize() {
027        if (!initialized) try {
028            daemon.setDaemon(true);
029            daemon.start();
030            
031            initialized = true;
032        } catch (Exception e) {
033            // Must have already started...
034        }
035    }
036
037    /** The monitoring code. Will be run in a separate thread. */
038    private static void monitor() {
039        while (true) synchronized (daemon) {
040            try {
041                daemon.wait(policy.heapAnalysisInterval);
042                
043                /**Capture system status*/
044                MemoryUtil.updateMemoryStatus();
045                
046            } catch (InterruptedException ie) {
047                // This should actually never happen...
048            } finally {
049                analyze();
050                System.gc();
051            }
052        }
053    }
054
055    /** Force early heap analysis. */
056    public static void force() {
057        synchronized (daemon) {
058            daemon.notifyAll();
059        };
060    }
061
062    /** Analyze memory usage and swap if necessary. */
063    private static void analyze() {
064        double target  = policy.fgHeapOptUsage;
065        double trigger = policy.fgHeapMaxUsage;
066
067        if (MemoryUtil.isBackgrounded()) {
068            target  = policy.bgHeapOptUsage;
069            trigger = policy.bgHeapMaxUsage;
070        }
071
072        if (MemoryUtil.heapUsage() > trigger)
073            swapUntil(target);
074    }
075
076    /**
077     * Swap until target utilization is reached (or there is nothing to swap).
078     */
079    private static void swapUntil(double target) {
080        while (MemoryUtil.heapUsage() > target &&
081               SwapReference.swapOutLeastUsed())
082            Thread.yield();  // Precaution - don't overutilize CPU.
083    }
084
085    private SwapManager() { /* Don't make me. */ }
086}
